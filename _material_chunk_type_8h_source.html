26 // Pack an eight character string into a 64 bit integer.
27 constexpr inline uint64_t charTo64bitNum(const char str[9]) noexcept {
28  return
29  ( (static_cast<uint64_t >(str[0]) << 56))
30  | ((static_cast<uint64_t >(str[1]) << 48) & 0x00FF000000000000U)
31  | ((static_cast<uint64_t >(str[2]) << 40) & 0x0000FF0000000000U)
32  | ((static_cast<uint64_t >(str[3]) << 32) & 0x000000FF00000000U)
33  | ((static_cast<uint64_t >(str[4]) << 24) & 0x00000000FF000000U)
34  | ((static_cast<uint64_t >(str[5]) << 16) & 0x0000000000FF0000U)
35  | ((static_cast<uint64_t >(str[6]) << 8) & 0x000000000000FF00U)
36  | ( static_cast<uint64_t >(str[7]) & 0x00000000000000FFU);
37 }
