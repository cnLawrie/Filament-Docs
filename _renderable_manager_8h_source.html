205 template<typename VECTOR, typename INDEX, typename, typename>
206 Box RenderableManager::computeAABB(VECTOR const* vertices, INDEX const* indices, size_t count,
207  size_t stride) noexcept {
208  math::float3 bmin(std::numeric_limits<float>::max());
209  math::float3 bmax(std::numeric_limits<float>::lowest());
210  for (size_t i = 0; i < count; ++i) {
211  VECTOR const* p = reinterpret_cast<VECTOR const*>(
212  (char const*)vertices + indices[i] * stride);
213  const math::float3 v(p->x, p->y, p->z);
214  bmin = min(bmin, v);
215  bmax = max(bmax, v);
216  }
217  return Box().set(bmin, bmax);
218 }
