60 /*
61  * Dispatcher is a data structure containing only function pointers.
62  * Each function pointer targets code that unpacks the arguments to the driver's method and
63  * calls it.
64  *
65  * Dispatcher's function pointers are populated during initialization and no CommandStream calls
66  * can be made before that.
67  *
68  * When a command is inserted into the stream, the corresponding function pointer is copied
69  * directly into CommandBase from Dispatcher.
70  */
71 class Dispatcher {
72 public:
73  using Execute = void (*)(Driver& driver, CommandBase* self, intptr_t* next);
74 #define DECL_DRIVER_API_SYNCHRONOUS(RetType, methodName, paramsDecl, params)
75 #define DECL_DRIVER_API(methodName, paramsDecl, params) Execute methodName##_;
76 #define DECL_DRIVER_API_RETURN(RetType, methodName, paramsDecl, params) Execute methodName##_;
77 #include "DriverAPI.inc"
78 };

82 class CommandBase {
83  static constexpr size_t FILAMENT_OBJECT_ALIGNMENT = alignof(std::max_align_t);
84 
85 protected:
86  using Execute = Dispatcher::Execute;
87 
88  constexpr explicit CommandBase(Execute execute) noexcept : mExecute(execute) {}
89 
90 public:
91  // alignment of all Commands in the CommandStream
92  static constexpr size_t align(size_t v) {
93  return (v + (FILAMENT_OBJECT_ALIGNMENT - 1)) & -FILAMENT_OBJECT_ALIGNMENT;
94  }
95 
96  // executes this command and returns the next one
97  inline CommandBase* execute(Driver& driver) {
98  // it is important to return the next command offset by output parameter instead
99  // of return value -- it allows the compiler to perform the tail call optimization.
100  intptr_t next;
101  mExecute(driver, this, &next);
102  return reinterpret_cast<CommandBase*>(reinterpret_cast<intptr_t>(this) + next);
103  }
104 
105  inline ~CommandBase() noexcept = default;
106 
107 private:
108  Execute mExecute;
109 };

113 template<typename M, typename D, typename T, std::size_t... I>
114 constexpr void trampoline(M&& m, D&& d, T&& t, std::index_sequence<I...>) {
115  (d.*m)(std::move(std::get<I>(std::forward<T>(t)))...);
116 }
117 
118 template<typename M, typename D, typename T>
119 constexpr void apply(M&& m, D&& d, T&& t) {
120  trampoline(std::forward<M>(m), std::forward<D>(d), std::forward<T>(t),
121  std::make_index_sequence< std::tuple_size<std::remove_reference_t<T>>::value >{});
122 }

124 /*
125  * CommandType<> is just a wrapper class to specialize on a pointer-to-member of Driver
126  * (i.e. a method pointer to a method of Driver of a particular type -- but not the
127  * method itself). We only do that so we can identify the parameters of that method.
128  * We won't call through that pointer though.
129  */
130 template<typename... ARGS>
131 struct CommandType;
132 
133 template<typename... ARGS>
134 struct CommandType<void (Driver::*)(ARGS...)> {
135 
136  /*
137  * Command is templated on a specific method of Driver, using CommandType's template
138  * parameter.
139  * Note that we're never calling this method (which is why it doesn't appear in the
140  * template parameter below). The actual call is made through Command::execute().
141  */
142  template<void(Driver::*)(ARGS...)>
143  class Command : public CommandBase {
144  // We use a std::tuple<> to record the arguments passed to the constructor
145  using SavedParameters = std::tuple<typename std::decay<ARGS>::type...>;
146  SavedParameters mArgs;
147 
148  void log() noexcept;
149  template<std::size_t... I> void log(std::index_sequence<I...>) noexcept;
150 
151  public:
152  template<typename M, typename D>
153  static inline void execute(M&& method, D&& driver, CommandBase* base, intptr_t* next) noexcept {
154  Command* self = static_cast<Command*>(base);
155  *next = align(sizeof(Command));
156  if (DEBUG_COMMAND_STREAM) {
157  // must call this before invoking the method
158  self->log();
159  }
160  apply(std::forward<M>(method), std::forward<D>(driver), self->mArgs);
161  self->~Command();
162  }
163 
164  // A command can be moved
165  inline Command(Command&& rhs) noexcept = default;
166 
167  template<typename... A>
168  inline explicit constexpr Command(Execute execute, A&& ... args)
169  : CommandBase(execute), mArgs(std::move(args)...) {
170  }
171 
172  // placement new declared as "throw" to avoid the compiler's null-check
173  inline void* operator new(std::size_t size, void* ptr) {
174  assert(ptr);
175  return ptr;
176  }
177  };
178 };

216 class CommandStream {
217 public:
218 #define DECL_DRIVER_API(methodName, paramsDecl, params) \
219  inline void methodName(paramsDecl) { \
220  DEBUG_COMMAND(methodName, params); \
221  using Cmd = COMMAND_TYPE(methodName); \
222  void* const p = allocateCommand(CommandBase::align(sizeof(Cmd))); \
223  new(p) Cmd(mDispatcher->methodName##_, params); \
224  }

226 #define DECL_DRIVER_API_SYNCHRONOUS(RetType, methodName, paramsDecl, params) \
227  inline RetType methodName(paramsDecl) { \
228  DEBUG_COMMAND(methodName, params); \
229  return mDriver->methodName(params); \
230  }
231 
232 #define DECL_DRIVER_API_RETURN(RetType, methodName, paramsDecl, params) \
233  inline RetType methodName(paramsDecl) { \
234  DEBUG_COMMAND(methodName, params); \
235  RetType result = mDriver->methodName##S(); \
236  using Cmd = COMMAND_TYPE(methodName##R); \
237  void* const p = allocateCommand(CommandBase::align(sizeof(Cmd))); \
238  new(p) Cmd(mDispatcher->methodName##_, RetType(result), params); \
239  return result; \
240  }

298 void* CommandStream::allocate(size_t size, size_t alignment) noexcept {
299  // make sure alignment is a power of two
300  assert(alignment && !(alignment & alignment-1));
301 
302  // pad the requested size to accommodate NoopCommand and alignment
303  const size_t s = CustomCommand::align(sizeof(NoopCommand) + size + alignment - 1);
304 
305  // allocate space in the command stream and insert a NoopCommand
306  char* const p = (char *)allocateCommand(s);
307  new(p) NoopCommand(p + s);
308 
309  // calculate the "user" data pointer
310  void* data = (void *)((uintptr_t(p) + sizeof(NoopCommand) + alignment - 1) & ~(alignment - 1));
311  assert(data >= p + sizeof(NoopCommand));
312  return data;
313 }
