29 namespace filament {
30 namespace backend {
31 
32 class Program {
33 public:
34 
35  static constexpr size_t SHADER_TYPE_COUNT = 2;
36  static constexpr size_t UNIFORM_BINDING_COUNT = CONFIG_UNIFORM_BINDING_COUNT;
37  static constexpr size_t SAMPLER_BINDING_COUNT = CONFIG_SAMPLER_BINDING_COUNT;
38 
39  enum class Shader : uint8_t {
40  VERTEX = 0,
41  FRAGMENT = 1
42  };
43 
44  struct Sampler {
45  utils::CString name = {}; // name of the sampler in the shader
46  size_t binding = 0; // binding point of the sampler in the shader
47  };
48 
49  using SamplerGroupInfo = std::array<std::vector<Sampler>, SAMPLER_BINDING_COUNT>;
50  using UniformBlockInfo = std::array<utils::CString, UNIFORM_BINDING_COUNT>;
51 
52  Program() noexcept;
53  Program(const Program& rhs) = delete;
54  Program& operator=(const Program& rhs) = delete;
55  Program(Program&& rhs) noexcept;
56  Program& operator=(Program&& rhs) noexcept;
57  ~Program() noexcept;
58 
59  // sets the material name and variant for diagnostic purposes only
60  Program& diagnostics(utils::CString const& name, uint8_t variantKey = 0);
61  Program& diagnostics(utils::CString&& name, uint8_t variantKey = 0) noexcept;
62 
63  // sets one of the program's shader (e.g. vertex, fragment)
64  Program& shader(Shader shader, void const* data, size_t size) noexcept;
65 
66  // sets the 'bindingPoint' uniform block's name for this program.
67  //
68  // Note: This is only needed for GLES3.0 backends, because the layout(binding=) syntax is
69  // not permitted in glsl. The backend needs a way to associate a uniform block
70  // to a binding point.
71  //
72  Program& setUniformBlock(size_t bindingPoint, utils::CString uniformBlockName) noexcept;
73 
74  // sets the 'bindingPoint' sampler group descriptor for this program.
75  // 'samplers' can be destroyed after this call.
76  // This effectively associates a set of (BindingPoints, index) to a texture unit in the shader.
77  // Or more precisely, what layout(binding=) is set to in GLSL.
78  Program& setSamplerGroup(size_t bindingPoint, Sampler const* samplers, size_t count) noexcept;
79 
80  Program& withVertexShader(void const* data, size_t size) {
81  return shader(Shader::VERTEX, data, size);
82  }
83 
84  Program& withFragmentShader(void const* data, size_t size) {
85  return shader(Shader::FRAGMENT, data, size);
86  }
87 
88  std::array<std::vector<uint8_t>, SHADER_TYPE_COUNT> const& getShadersSource() const noexcept {
89  return mShadersSource;
90  }
91 
92  UniformBlockInfo const& getUniformBlockInfo() const noexcept { return mUniformBlocks; }
93 
94  SamplerGroupInfo const& getSamplerGroupInfo() const { return mSamplerGroups; }
95 
96  const utils::CString& getName() const noexcept { return mName; }
97 
98  uint8_t getVariant() const noexcept { return mVariant; }
99 
100  bool hasSamplers() const noexcept { return mHasSamplers; }
101 
102 private:
103 #if !defined(NDEBUG)
104  friend utils::io::ostream& operator<< (utils::io::ostream& out, const Program& builder);
105 #endif
106 
107  // FIXME: none of these fields should be public as this is a public API
108  UniformBlockInfo mUniformBlocks = {};
109  SamplerGroupInfo mSamplerGroups = {};
110  std::array<std::vector<uint8_t>, SHADER_TYPE_COUNT> mShadersSource;
111  utils::CString mName;
112  bool mHasSamplers = false;
113  uint8_t mVariant;
114 };
115 
116 } // namespace backend;
117 } // namespace filament;
