179 // converts an RGBA color to linear space
180 // the conversion depends on the specified type
181 inline LinearColorA Color::toLinear(RgbaType type, math::float4 color) {
182  switch (type) {
183  case RgbaType::sRGB:
184  return Color::toLinear<ACCURATE>(color) * math::float4{color.a, color.a, color.a, 1.0f};
185  case RgbaType::LINEAR:
186  return color * math::float4{color.a, color.a, color.a, 1.0f};
187  case RgbaType::PREMULTIPLIED_sRGB:
188  return Color::toLinear<ACCURATE>(color);
189  case RgbaType::PREMULTIPLIED_LINEAR:
190  return color;
191  }
192 }
