44 class HandleBase {
45 public:
46  using HandleId = uint32_t;
47  static constexpr const HandleId nullid = HandleId{ std::numeric_limits<HandleId>::max() };
48 
49  enum class no_init { };
50  static constexpr no_init NO_INIT = { };
51 
52  constexpr HandleBase() noexcept : object(nullid) { }
53 
54  explicit HandleBase(no_init) noexcept { } // NOLINT
55 
56  explicit HandleBase(HandleId id) noexcept : object(id) {
57  assert(object != nullid); // usually means an uninitialized handle is used
58  }
59 
60  HandleBase(HandleBase const& rhs) noexcept = default;
61 
62  HandleBase& operator = (HandleBase const& rhs) noexcept = default;
63 
64  HandleBase(HandleBase&& rhs) noexcept : object(rhs.object) {
65  rhs.object = nullid;
66  }
67 
68  HandleBase& operator = (HandleBase&& rhs) noexcept {
69  std::swap(object, rhs.object);
70  return *this;
71  }
72 
73  void clear() noexcept { object = nullid; }
74 
75  explicit operator bool() const noexcept { return object != nullid; }
76 
77  bool operator==(const HandleBase& rhs) const noexcept { return object == rhs.object; }
78  bool operator!=(const HandleBase& rhs) const noexcept { return object != rhs.object; }
79 
80  // get this handle's handleId
81  HandleId getId() const noexcept { return object; }
82 
83 protected:
84  HandleId object;
85 };
