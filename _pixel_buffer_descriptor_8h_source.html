65  static constexpr size_t computeDataSize(PixelDataFormat format, PixelDataType type,
66  size_t stride, size_t height, size_t alignment) noexcept {
67  assert(alignment);
68 
69  if (type == PixelDataType::COMPRESSED) {
70  return 0;
71  }
72 
73  size_t n = 0;
74  switch (format) {
75  case PixelDataFormat::R:
76  case PixelDataFormat::R_INTEGER:
77  case PixelDataFormat::DEPTH_COMPONENT:
78  case PixelDataFormat::ALPHA:
79  n = 1;
80  break;
81  case PixelDataFormat::RG:
82  case PixelDataFormat::RG_INTEGER:
83  case PixelDataFormat::DEPTH_STENCIL:
84  n = 2;
85  break;
86  case PixelDataFormat::RGB:
87  case PixelDataFormat::RGB_INTEGER:
88  n = 3;
89  break;
90  case PixelDataFormat::UNUSED: // shouldn't happen (used to be rgbm)
91  case PixelDataFormat::RGBA:
92  case PixelDataFormat::RGBA_INTEGER:
93  n = 4;
94  break;
95  }
96 
97  size_t bpp = n;
98  switch (type) {
99  case PixelDataType::COMPRESSED: // Impossible -- to squash the IDE warnings
100  case PixelDataType::UBYTE:
101  case PixelDataType::BYTE:
102  // nothing to do
103  break;
104  case PixelDataType::USHORT:
105  case PixelDataType::SHORT:
106  case PixelDataType::HALF:
107  bpp *= 2;
108  break;
109  case PixelDataType::UINT:
110  case PixelDataType::INT:
111  case PixelDataType::FLOAT:
112  bpp *= 4;
113  break;
114  case PixelDataType::UINT_10F_11F_11F_REV:
115  // Special case, format must be RGB and uses 4 bytes
116  assert(format == PixelDataFormat::RGB);
117  bpp = 4;
118  break;
119  }
120 
121  size_t bpr = bpp * stride;
122  size_t bprAligned = (bpr + (alignment - 1)) & -alignment;
123  return bprAligned * height;
124  }

126  uint32_t left = 0;
127  uint32_t top = 0;

134  uint32_t imageSize;

136  };
137  };
138  PixelDataType type : 4;
139  uint8_t alignment : 4;
140 };
